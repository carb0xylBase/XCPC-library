#include <bits/stdc++.h>
using namespace std;

int main() {
    srand(time(0));
    ofstream fout("input.txt");

    int T = 5; // 最大数据组数
    fout << T << "\n";

    for (int t = 0; t < T; ++t) {
        int n = 100000; // 最大国家数
        int q = 100000; // 最大修改次数
        fout << n << " " << q << "\n";

        // 生成长度为 n 的随机 0/1 字符串
        for (int i = 0; i < n; ++i) {
            fout << (rand() % 2);
        }
        fout << "\n";

        // 生成 q 条随机修改操作
        for (int i = 0; i < q; ++i) {
            int op = (rand() % 2) + 1; // 1 或 2
            int l = rand() % n + 1;
            int r = rand() % n + 1;
            if (l > r) swap(l, r);
            fout << op << " " << l << " " << r << "\n";
        }
    }

    fout.close();
    return 0;
}
